58 
59  enum MailboxType mb_type = mx_path_probe(mailbox_path(m_check));
60 
61  const bool c_mail_check_recent =
62  cs_subset_bool(NeoMutt->sub, "mail_check_recent");
63  if ((m_cur == m_check) && c_mail_check_recent)
64  m_check->has_new = false;
65 
66  switch (mb_type)
67  {
68  case MUTT_POP:
69  case MUTT_NNTP:
70  case MUTT_NOTMUCH:
71  case MUTT_IMAP:
72  m_check->type = mb_type;
73  break;
74  default:
75  if ((stat(mailbox_path(m_check), &sb) != 0) ||
76  ((m_check->type == MUTT_UNKNOWN) && S_ISREG(sb.st_mode) && (sb.st_size == 0)) ||
77  ((m_check->type == MUTT_UNKNOWN) &&
78  ((m_check->type = mx_path_probe(mailbox_path(m_check))) <= 0)))
79  {
80  /* if the mailbox still doesn't exist, set the newly created flag to be
81  * ready for when it does. */
82  m_check->newly_created = true;
83  m_check->type = MUTT_UNKNOWN;
84  m_check->size = 0;
85  return;
86  }
87  break; // kept for consistency.
88  }
89 
90  const bool c_check_mbox_size =
91  cs_subset_bool(NeoMutt->sub, "check_mbox_size");
92  /* check to see if the folder is the currently selected folder before polling */
93  if (!m_cur || mutt_buffer_is_empty(&m_cur->pathbuf) ||
94  (((m_check->type == MUTT_IMAP) || (m_check->type == MUTT_NNTP) ||
95  (m_check->type == MUTT_NOTMUCH) || (m_check->type == MUTT_POP)) ?
96  !mutt_str_equal(mailbox_path(m_check), mailbox_path(m_cur)) :
97  ((sb.st_dev != ctx_sb->st_dev) || (sb.st_ino != ctx_sb->st_ino))))
98  {
99  switch (m_check->type)
100  {
101  case MUTT_IMAP:
102  case MUTT_MBOX:
103  case MUTT_MMDF:
104  case MUTT_MAILDIR:
105  case MUTT_MH:
106  case MUTT_NOTMUCH:
107  if ((mx_mbox_check_stats(m_check, check_stats) != MX_STATUS_ERROR) &&
108  m_check->has_new)
109  {
110  MailboxCount++;
111  }
112  break;
113  default:; /* do nothing */
114  }
115  }
116  else if (c_check_mbox_size && m_cur && mutt_buffer_is_empty(&m_cur->pathbuf))
117  m_check->size = (off_t) sb.st_size; /* update the size of current folder */
118 
119  if (!m_check->has_new)
120  m_check->notified = false;
121  else if (!m_check->notified)
122  MailboxNotify++;
123 }
