701  * resorting, so we sort backwards and then put them back
702  * in reverse order so they're forwards */

731  thread->sort_key = thread->message;

733  if (thread->next)
734  {
735  thread = thread->next;
736  continue;
737  }
738  }

899  top.prev = NULL;

906  * exists. otherwise, if there is a MuttThread that already has a message, thread

910  cs_subset_bool(NeoMutt->sub, "duplicate_threads");
911  for (i = 0; i < m->msg_count; i++)
912  {

915  continue;

1000  }
1001  }

1027  else

1038  * the second reference (since at least eudora puts the most

1040  if (STAILQ_EMPTY(&e->env->references))

1075  thread = tnew;

1077  break;
1078  }
1079

1081  insert_message(&top.child, &top, thread);
1082  }
1083

1085  for (thread = top.child; thread; thread = thread->next)
1086  {
1087  thread->parent = NULL;
1088  }
1089  tctx->tree = top.child;
1090
