668 {

677  *size = 256;
678  line = mutt_mem_malloc(*size);
679  }

682  {

711  {
712  /* The last line of fp isn't \n terminated */
713  if (line_num)
714  (*line_num)++;

721  offset = *size - 1; /* overwrite the terminating 0 */
